SELECT 
  SUM(TRABALHO.TRABALHO_TOTAL) AS TRABALHO_TOTAL,
  TIPO_EMPREENDIMENTO.DESCRICAO AS TIPO_EMPREENDIMENTO,
  COUNT(DEMANDA.ID) AS NUMERO_DEMANDAS
FROM
  TRABALHO
  INNER JOIN DEMANDA ON (TRABALHO.DEMANDA = DEMANDA.ID)
  INNER JOIN EMPREENDIMENTO ON (DEMANDA.EMPREENDIMENTO = EMPREENDIMENTO.ID)
  INNER JOIN TIPO_EMPREENDIMENTO ON (EMPREENDIMENTO.TIPO_EMPREENDIMENTO = TIPO_EMPREENDIMENTO.ID)
GROUP BY
  TIPO_EMPREENDIMENTO.DESCRICAO,
  DEMANDA.ID
ORDER BY
  TIPO_EMPREENDIMENTO.DESCRICAO

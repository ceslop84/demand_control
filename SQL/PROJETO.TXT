SELECT 
  SUM(TRABALHO.TRABALHO_TOTAL) AS TRABALHO_TOTAL,
  PROJETO.DESCRICAO AS PROJETO,
  COUNT(DEMANDA.ID) AS NUMERO_DEMANDAS
FROM
  TRABALHO
  INNER JOIN DEMANDA ON (TRABALHO.DEMANDA = DEMANDA.ID)
  INNER JOIN PROJETO ON (DEMANDA.PROJETO = PROJETO.ID)
GROUP BY
  PROJETO.DESCRICAO
ORDER BY
  PROJETO.DESCRICAO
